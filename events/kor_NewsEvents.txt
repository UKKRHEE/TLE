news_event = { #파나이 호 사건
	id = kor_news.6
	title = kor_news.6.t
	desc = kor_news.6.desc
	picture = GFX_news_event_panay_incident
	
	major = yes
	fire_only_once = yes	
    trigger = {
		KOR = {
			controls_state = 608
			controls_state = 826
			controls_state = 829
			KOR = {
				OR = {
					has_war_with = CHI
					has_war_with = ZLC
					has_war_with = WZC
				}
				NOT = {
					has_government = democratic
					has_government = communism
				}
			}
		}
		date > 1940.8.1
	}
	mean_time_to_happen = {
		days = 14
	}
	immediate = {
		USA = {
			country_event = {
				id = korea.63
			}
		}
		KOR = {
			country_event = {
				id = korea.64
			}
			set_country_flag = KOR_PANAI_PROBLEM
		}
	}		
	
	
	option = {
		name = kor_news.6.a
		trigger = {
			TAG = KOR
		}
		unlock_decision_tooltip = KOR_confiscation_of_US_assets
		unlock_decision_tooltip = KOR_not_say_the_panay
		unlock_decision_tooltip = KOR_hunlian_in_pacific
	}
	option = {
		name = kor_news.6.b
		trigger = {
			TAG = USA
		}
	}
	option = {
		name = kor_news.6.c
		trigger = {
			NOT = {
				OR = {
					TAG = KOR
					TAG = USA
				}
			}			
		}
		USA = {
			add_opinion_modifier = { target = KOR modifier = panay_inc2_opinion_modifier }
		}
		ENG = {
			add_opinion_modifier = { target = KOR modifier = panay_inc2_opinion_modifier }
		}
	}
}
